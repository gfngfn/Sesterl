
letrec wait_all(n) =
  if n <= 0 then
    return(())
  else
    receive
    | x ->
        let u = print_debug(x) in
        wait_all(n - 1)
    end


letrec aux(n) =
  if n <= 0 then
    return(())
  else
    do parent <- self() in
    do pid <-
      spawn(
        send(parent, n)
      )
    in
    aux(n - 1)


let main() =
  let m = 10 in
  do aux(m) in
  wait_all(m)
